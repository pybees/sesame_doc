    [docs]class Dipole(object):
    """Single current dipole class for SESAME.

    Parameters
    ----------
    loc : :py:class:`int`
        The dipole location (as an index of a brain grid).
    """
    def __init__(self, loc):
        self.loc = loc

    def __repr__(self):
        s = 'location : {0}'.format(self.loc)

        return '<Dipole  |  {0}>'.format(s)
